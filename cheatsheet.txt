seed:
npx prisma db seed

prereq for seeding:
npm install @prisma/adapter-pg
npm install --save-dev @types/pg
npm install -D tsx

RUN FROM PRISMA FOLDER:

npx prisma generate

migrations:
- reset DB and apply all migrations:
 npx prisma migrate reset --config ../prisma.config.ts

- apply all existing migrations:
npx prisma migrate deploy --config ../prisma.config.ts

- apply new migration:
npx prisma migrate dev --name <migration_name>  --config ../prisma.config.ts
